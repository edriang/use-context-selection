```bash
npm install --save use-context-selection
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'use-context-selection'
import 'use-context-selection/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
